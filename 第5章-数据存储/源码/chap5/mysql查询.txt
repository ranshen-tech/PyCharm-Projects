单表查询
(1)查询表中的全部数据
  select * from emp;
(2)查询表中的部分数据
   a:部分列
   查询所有的员工的编号，姓名，薪水
   select empno,ename,sal from emp;

   b:部分行
    查询薪水大于2500的员工信息
    select * from emp where sal>2500;

   c:查询部分行和列
    查询薪水大于2500的员工编号，姓名,薪水
    select empno,ename,sal from emp where sal>2500;

(3)查询结果的排序
  select * from emp order by sal asc;
  select * from emp order by sal desc;

 (4)模糊查询
   a:  like   像
    查询员工姓名中含有M 的员工信息
    select * from emp where ename like '%M%';
    查询员工姓名以M开头的员工信息
    select * from emp where ename like 'M%';
    查询员工姓名中第二个字母是M的员工信息
    select * from emp where ename like '_M%';

   b:区间范围  between ...and
    查询员工的薪水在2000到3000之间的员工信息
    select * from emp where sal between 2000 and 3000 order by sal desc;
   c:在指定的值中进行选择  in
    查询职位为 'CLERK','MANAGER' 的员工信息
    select * from emp where job in ('CLERK','MANAGER')order by job ;

  (5)分组函数，带函数的查询
    查询表中的所有行和列
    select * from emp

    查询一共有多少员工?
    select count(*) from emp;
    select count(1) from emp;
    select count(*) from emp where deptno=20;
    select count(*) from emp where job='CLERK';

    查询员工的总薪水或平均薪水
    select sum(sal) from emp;
    select sum(sal) from emp where deptno=20;
    select sum(sal) from emp where job='CLERK';

    select avg(sal) from emp;
    select avg(sal) from emp where deptno=20;
    select avg(sal) from emp where job='CLERK';

    select max(sal) from emp;
    select max(sal) from emp where deptno=20;
    select max(sal) from emp where job='CLERK';

    select min(sal) from emp;
    select min(sal) from emp where deptno=20;
    select min(sal) from emp where job='CLERK';

  (6)分组查询 group by ...having  与分组函数count,max,min,avg,sum一起使用
    查询每个部分的总人数
    select deptno,count(*) from emp group by deptno;

    查询每个部分的总人数不低于4人的部门编号及部门总人数
    select deptno,count(*) from emp group by deptno having count(*)>3;


  (7)表连接查询
    99标准的表连接查询
    查询员工的编号，姓名，薪水，员工的部门名称
    select empno,ename,sal,dname from emp  inner join dept on emp.deptno=dept.deptno where sal>2000 order by sal desc ;

    查询所有部门员工的编号，姓名，薪水，部门名称  ，包含没有员工的部门
    select empno,ename,sal,dname from emp  right join dept on emp.deptno=dept.deptno;

    select empno,ename,sal,dname from dept  left join emp on emp.deptno=dept.deptno;
